Example

{
  "global": {
    "baseUrl": "https://api.example.com",
    "maxRequests": 200,
    "concurrent": 10,
    "timeout": 15000,
    "executor": "arrival-rate",
    "arrivalRate": 30,
    "reporters": [
      { "type": "json", "path": "./out/result.json" },
      { "type": "html", "path": "./out/result.html" }
    ]
  }
}
